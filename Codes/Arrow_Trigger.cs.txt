using UnityEngine;

public class Arrow_Trigger : MonoBehaviour
{
    public string direction;                
    public VRNavigator navigator;       
    void Start()
    {

    }
    public void OnMouseDown(){
        if (navigator != null){
            navigator.MoveInDirection(direction);
        }
        else{
            Debug.LogError("Navigator not set on ArrowTrigger.");
        }
    }
}
